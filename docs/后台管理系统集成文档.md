# 音乐榜单可视化平台 — 后台管理系统集成文档（DVAdmin）

## 总体设计思路

DVAdmin（django-vue3-admin）是一套完整的前后端分离管理框架，
**采用"嵌入式集成"策略**：将 DVAdmin 作为独立目录 `admin/` 放在项目根目录，
后端在 DVAdmin 的 Django 工程中注册一个自定义 App `music`，
使用 `managed = False` 的 Django ORM 模型映射已有 MySQL 表（不重建表），
通过 DVAdmin 的 `CustomModelViewSet` 快速暴露 CRUD 接口，
前端在 DVAdmin 的 Vue3 工程中添加音乐管理页面。

---

## 最终目录结构

```
music-analysis/
├── frontend/          ← 现有大屏前端（Vue2，端口 8080）
├── backend/           ← 现有大屏后端（Django，端口 8000）
├── crawler/           ← 爬虫模块
├── docs/              ← 文档
└── admin/             ← 新增：后台管理系统（DVAdmin）
    ├── backend/       ← DVAdmin Django 后端（端口 8001）
    │   ├── application/   ← Django 工程配置
    │   ├── conf/          ← 环境配置
    │   ├── dvadmin/       ← DVAdmin 系统模块（不动）
    │   ├── apps/          ← 自定义业务 App（新增）
    │   │   └── music/     ← 音乐管理 App（新增）
    │   ├── manage.py
    │   └── requirements.txt
    └── web/           ← DVAdmin Vue3 前端（端口 8081）
        └── src/
            └── views/
                └── music/  ← 音乐管理页面（新增）
```

## 第四步：定义 ORM 模型（映射已有表）

`managed = False` 表示 Django 只读取表结构、不创建也不修改表，完全安全。

```python
# admin/backend/apps/music/models.py

from django.db import models
from dvadmin.utils.models import CoreModel


# ================================================================
# 维度表：歌手
# ================================================================
class DimSinger(models.Model):
    singer_id   = models.CharField(primary_key=True, max_length=50, verbose_name='歌手ID')
    singer_name = models.CharField(max_length=100, verbose_name='歌手名称')
    created_at  = models.DateTimeField(auto_now_add=True, verbose_name='创建时间')

    class Meta:
        managed    = False          # 不让 Django 管理这张表
        db_table   = 'dim_singer'
        verbose_name = '歌手信息'

    def __str__(self):
        return self.singer_name


# ================================================================
# 维度表：周榜期数
# ================================================================
class DimWeekIssue(models.Model):
    issue        = models.IntegerField(primary_key=True, verbose_name='期数编号')
    month        = models.CharField(max_length=20, verbose_name='月份')
    start_date   = models.DateField(null=True, blank=True, verbose_name='开始日期')
    end_date     = models.DateField(null=True, blank=True, verbose_name='结束日期')
    title        = models.CharField(max_length=100, verbose_name='标题')
    publish_time = models.DateTimeField(null=True, blank=True, verbose_name='发布时间')

    class Meta:
        managed    = False
        db_table   = 'dim_week_issue'
        verbose_name = '周榜期数'
        ordering   = ['-issue']

    def __str__(self):
        return f"第{self.issue}期 {self.title}"


# ================================================================
# 事实表：历史周榜明细
# ================================================================
class FactWeeklyChart(models.Model):
    id          = models.AutoField(primary_key=True)
    issue       = models.IntegerField(verbose_name='周榜期数')
    rank        = models.IntegerField(verbose_name='排名')
    song_id     = models.CharField(max_length=50, verbose_name='歌曲ID')
    song_name   = models.CharField(max_length=200, verbose_name='歌曲名称')
    singer_id   = models.CharField(max_length=50, verbose_name='歌手ID')
    singer_name = models.CharField(max_length=100, verbose_name='歌手名称')
    uni_index   = models.DecimalField(max_digits=8, decimal_places=2, null=True, verbose_name='综合指数')
    new_flag    = models.BooleanField(default=False, verbose_name='是否新入榜')
    cover_image = models.TextField(null=True, blank=True, verbose_name='封面图URL')

    class Meta:
        managed     = False
        db_table    = 'fact_weekly_chart'
        verbose_name = '历史周榜'
        ordering    = ['-issue', 'rank']

    def __str__(self):
        return f"[第{self.issue}期 #{self.rank}] {self.song_name}"


# ================================================================
# 事实表：今日日榜
# ================================================================
class FactDailyChart(models.Model):
    id            = models.AutoField(primary_key=True)
    chart_date    = models.DateField(verbose_name='榜单日期')
    rank          = models.IntegerField(verbose_name='排名')
    song_id       = models.CharField(max_length=50, verbose_name='歌曲ID')
    song_name     = models.CharField(max_length=200, verbose_name='歌曲名称')
    singer_id     = models.CharField(max_length=50, null=True, blank=True, verbose_name='歌手ID')
    singer_name   = models.CharField(max_length=100, verbose_name='歌手名称')
    score         = models.DecimalField(max_digits=8, decimal_places=2, null=True, verbose_name='得分')
    incr_rank     = models.IntegerField(default=0, verbose_name='排名变化')
    incr_score    = models.DecimalField(max_digits=8, decimal_places=2, default=0, verbose_name='得分变化')
    play_index    = models.DecimalField(max_digits=8, decimal_places=2, null=True, verbose_name='播放指数')
    pay_index     = models.DecimalField(max_digits=8, decimal_places=2, null=True, verbose_name='畅销指数')
    pop_index     = models.DecimalField(max_digits=8, decimal_places=2, null=True, verbose_name='人气指数')
    days_on_chart = models.IntegerField(default=0, verbose_name='上榜天数')
    new_flag      = models.BooleanField(default=False, verbose_name='是否新入榜')
    track_tags    = models.JSONField(null=True, blank=True, verbose_name='标签')
    cover_image   = models.TextField(null=True, blank=True, verbose_name='封面图URL')

    class Meta:
        managed     = False
        db_table    = 'fact_daily_chart'
        verbose_name = '今日日榜'
        ordering    = ['-chart_date', 'rank']

    def __str__(self):
        return f"[{self.chart_date} #{self.rank}] {self.song_name}"


# ================================================================
# 聚合表：歌手全量统计
# ================================================================
class AggSingerTotal(models.Model):
    singer_id      = models.CharField(primary_key=True, max_length=50, verbose_name='歌手ID')
    singer_name    = models.CharField(max_length=100, verbose_name='歌手名称')
    total_count    = models.IntegerField(default=0, verbose_name='总上榜次数')
    champion_count = models.IntegerField(default=0, verbose_name='夺冠次数')
    top3_count     = models.IntegerField(default=0, verbose_name='Top3次数')
    last_updated   = models.DateTimeField(auto_now=True, verbose_name='更新时间')

    class Meta:
        managed     = False
        db_table    = 'agg_singer_total'
        verbose_name = '歌手统计'
        ordering    = ['-total_count']

    def __str__(self):
        return f"{self.singer_name}（{self.total_count}次）"


# ================================================================
# 聚合表：歌曲留榜生命力
# ================================================================
class AggSongLongevity(models.Model):

    song_name         = models.CharField(max_length=200, verbose_name='歌曲名称')

        db_table    = 'agg_song_longevity'
```

## 第六步：定义 ViewSet

DVAdmin 的 `CustomModelViewSet` 已内置分页、过滤、导出、权限控制，直接继承即可。

```python
# admin/backend/apps/music/views.py

from dvadmin.utils.viewset import CustomModelViewSet
from apps.music.models import (
    DimSinger, DimWeekIssue, FactWeeklyChart,
    FactDailyChart, AggSingerTotal, AggSongLongevity
)
from apps.music.serializers import (
    DimSingerSerializer, DimWeekIssueSerializer,
    FactWeeklyChartSerializer, FactDailyChartSerializer,
    AggSingerTotalSerializer, AggSongLongevitySerializer
)


class DimSingerViewSet(CustomModelViewSet):
    """歌手管理"""
    queryset         = DimSinger.objects.all()
    serializer_class = DimSingerSerializer
    search_fields    = ['singer_name', 'singer_id']
    ordering_fields  = ['singer_id']


class DimWeekIssueViewSet(CustomModelViewSet):
    """周榜期数管理"""
    queryset         = DimWeekIssue.objects.all()
    serializer_class = DimWeekIssueSerializer
    search_fields    = ['title', 'month']
    ordering_fields  = ['-issue']


class FactWeeklyChartViewSet(CustomModelViewSet):
    """历史周榜数据"""
    queryset         = FactWeeklyChart.objects.all()
    serializer_class = FactWeeklyChartSerializer
    search_fields    = ['song_name', 'singer_name']
    filterset_fields = ['issue', 'rank', 'singer_id', 'new_flag']
    ordering_fields  = ['-issue', 'rank']


class FactDailyChartViewSet(CustomModelViewSet):
    """今日日榜数据"""
    queryset         = FactDailyChart.objects.all()
    serializer_class = FactDailyChartSerializer
    search_fields    = ['song_name', 'singer_name']
    filterset_fields = ['chart_date', 'rank', 'new_flag']
    ordering_fields  = ['-chart_date', 'rank']


class AggSingerTotalViewSet(CustomModelViewSet):
    """歌手上榜统计"""
    queryset         = AggSingerTotal.objects.all()
    serializer_class = AggSingerTotalSerializer
    search_fields    = ['singer_name']
    ordering_fields  = ['-total_count']


class AggSongLongevityViewSet(CustomModelViewSet):
    """歌曲留榜统计"""
    queryset         = AggSongLongevity.objects.all()
    serializer_class = AggSongLongevitySerializer
    search_fields    = ['song_name', 'singer_name']
    ordering_fields  = ['-total_weeks']
```

## 第十二步：完整启动流程

```bash
# 1. 启动 DVAdmin 后端（端口 8001）
cd admin/backend
python manage.py runserver 0.0.0.0:8001

# 2. 启动 DVAdmin 前端（端口 8081）
cd admin/web
npm run dev

# 3. 启动现有大屏后端（保持原有，端口 8000）
cd backend/music_backend
python manage.py runserver 0.0.0.0:8000

# 4. 启动现有大屏前端（端口 8080）
cd frontend
npm run serve
```

访问入口：
- 大屏可视化：`http://localhost:8080`
- 后台管理系统：`http://localhost:8081`（账号 `superadmin` / `admin123456`）
- DVAdmin 后端 API：`http://localhost:8001/api/`
- API 文档（Swagger）：`http://localhost:8001/swagger/`
